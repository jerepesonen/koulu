SELECT DEPT.DEPT, DEPT.DUTY, EMPS.GNAME, EMPS.SURNAME 
FROM DEPT INNER JOIN EMPS
ON DEPT.DEPT = EMPS.DEPT
WHERE EMPS.DEPT =
	(SELECT EMPS.DEPT 
    FROM EMPS
    WHERE EMPS.GNAME = 'Peter' AND EMPS.SURNAME = 'CURRY');
    
SELECT EMPS.GNAME, EMPS.SURNAME, EMPS.PHONE, EMPS.DEPT 
FROM EMPS INNER JOIN DEPT
ON EMPS.DEPT = DEPT.DEPT
WHERE DEPT.DUTY LIKE 'MARKETING' OR DEPT.DEPT = (SELECT DEPT.DEPT FROM DEPT WHERE DEPT.MANAGER = 'Mark G');
